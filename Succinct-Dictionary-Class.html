class (Select0 t, Select1 t, Dictionary Bool t) => Ranked t whereSource

a classic bit-vector-based succinct indexed dictionary

For 1 <= i <= size t: 

 rank0 t i + rank1 t i = i

Given i = select1 t j 

 rank0 t i = i - j
 rank1 t i = j

Given i = select0 t j 

 rank0 t i = j
 rank1 t i = i - j

Methods

rank0 :: Ranked t => t -> Int -> IntSource

 rank0 t i = i - rank1 t i
 rank0 = rank False

rank1 :: Ranked t => t -> Int -> IntSource

 rank1 t i = i - rank0 t i
 rank1 = rank True

rank_ :: Ranked t => t -> Int -> IntSource

rank_ t i return the number of bits to the left of position i

When i > 1: 

 rank_ t i = rank1 t (i - 1)

The result is 0 otherwise.
