rank :: Elem t -> t -> Int -> IntSource

rank a t i returns the # of occurences of a in t up to position i for 0 < i <= size t

select :: Elem t -> t -> Int -> IntSource

select a t i returns the position of the ith appearance of a in t. as long as 0 < i <= rank a t (size t)
